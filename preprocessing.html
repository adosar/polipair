# Define the function that returns the number of k-best matches.
def kbest(X, Y, k=5, metric='cosine'):
    r"""
    Return the number of correct k-best matches.

    Parameters
    ----------
    X : array_like of shape (n_samples, n_features)
    Y : array_like of shape (n_samples, n_features)
    k : int, default=5
    metric : {'cosine', 'dot', 'euclidean'}, default='cosine'
        The similarity metric.

    Returns
    -------
    count : int
        The number of correct k-best matches.
    """
    distance_metrics = {
        'cosine': cosine_similarity,
        'dot' : linear_kernel,
        'euclidean': euclidean_distances,
    }

    distance_func = distance_metrics[metric]
    similarity = distance_func(X=X, Y=Y)

    # Negate the array to get sorting in descending order.
    order = 1 if metric == 'euclidean' else -1
    similarity_kbest = np.argsort(order * similarity, axis=1)[:, :k]

    # Create dummy index to match compared to similarity_kbest.
    index = np.tile(np.arange(len(similarity_kbest)), reps=(k, 1)).T

    return (index == similarity_kbest).sum()
